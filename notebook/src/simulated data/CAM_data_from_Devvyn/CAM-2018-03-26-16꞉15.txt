DROP TABLE IF EXISTS observer;
CREATE TABLE observer ( id PRIMARY KEY, parentID, firstName NOT NULL, lastName NOT NULL, active BOOLEAN NOT NULL );
DROP TABLE IF EXISTS client;
CREATE TABLE client ( id PRIMARY KEY, parentID, firstName NOT NULL, lastName NOT NULL, businessName , active BOOLEAN NOT NULL );
DROP TABLE IF EXISTS field;
CREATE TABLE field ( id PRIMARY KEY, parentID REFERENCES client, name NOT NULL, cropTypeID NOT NULL, description , date DATETIME NOT NULL );
DROP TABLE IF EXISTS observationSet;
CREATE TABLE observationSet ( id PRIMARY KEY, parentID REFERENCES field, cropStageID INTEGER NOT NULL, observerID NOT NULL REFERENCES observer, time DATETIME NOT NULL );
DROP TABLE IF EXISTS observationPoint;
CREATE TABLE observationPoint ( id PRIMARY KEY, parentID REFERENCES observationSet, spot INTEGER NOT NULL, latitude DOUBLE , longitude DOUBLE );
DROP TABLE IF EXISTS observation;
CREATE TABLE observation ( id PRIMARY KEY, parentID REFERENCES observationPoint, specieType INTEGER NOT NULL, observationIndex INTEGER NOT NULL );
DROP TABLE IF EXISTS observationSpecie;
CREATE TABLE observationSpecie ( id PRIMARY KEY, parentID REFERENCES observation, specieType INTEGER NOT NULL, specieID NOT NULL, count INTEGER NOT NULL );
